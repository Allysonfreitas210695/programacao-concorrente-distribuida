Questão 10 - MPI_Type_indexed (Triangular Superior)
====================================================

MPI_Type_indexed permite selecionar blocos de tamanhos e deslocamentos arbitrários
de um array, diferente de MPI_Type_create_struct os deslocamentos são em unidades
do tipo base (MPI_INT aqui), não em bytes.

Para uma matriz n×n em row-major (vetor 1D de tamanho n*n):
  - Linha 0: elementos 0..n-1         → blocklength=n,   displacement=0
  - Linha 1: elementos n+1..n+n-1     → blocklength=n-1, displacement=n+1
  - Linha i: elementos (i*n+i)..(i*n+n-1) → blocklength=n-i, displacement=i*n+i
  - ...
  - Linha n-1: elemento (n-1)*n+(n-1) → blocklength=1,   displacement=(n-1)*n+(n-1)

Exemplo 4×4: elementos 0,1,2,3,5,6,7,10,11,15

Trecho mais importante:
-----------------------
    for (int i = 0; i < n; i++) {
        blocklengths[i]  = n - i;       /* elementos na linha i acima da diagonal */
        displacements[i] = i * n + i;   /* offset em unidades de MPI_INT */
    }

    MPI_Datatype upper_tri_type;
    MPI_Type_indexed(n, blocklengths, displacements, MPI_INT, &upper_tri_type);
    MPI_Type_commit(&upper_tri_type);

    /* Uma única chamada envia toda a triangular superior */
    MPI_Send(matrix, 1, upper_tri_type, 1, 0, MPI_COMM_WORLD);

Código completo:
-----------------
/*
 * Compile: mpicc -g -Wall -o mpi_upper_q10 mpi_upper_q10.c
 * Run:     mpiexec -n 2 ./mpi_upper_q10
 */
#include <stdio.h>
#include <stdlib.h>
#include <mpi.h>

int main(void) {
    int my_rank, comm_sz;
    MPI_Init(NULL, NULL);
    MPI_Comm_rank(MPI_COMM_WORLD, &my_rank);
    MPI_Comm_size(MPI_COMM_WORLD, &comm_sz);

    int n;
    int *matrix = NULL;

    if (my_rank == 0) {
        printf("Digite n (ordem da matriz): ");
        scanf("%d", &n);
    }
    MPI_Bcast(&n, 1, MPI_INT, 0, MPI_COMM_WORLD);

    int upper_count = n * (n + 1) / 2;

    int *blocklengths  = malloc(n * sizeof(int));
    int *displacements = malloc(n * sizeof(int));

    for (int i = 0; i < n; i++) {
        blocklengths[i]  = n - i;
        displacements[i] = i * n + i;
    }

    MPI_Datatype upper_tri_type;
    MPI_Type_indexed(n, blocklengths, displacements, MPI_INT, &upper_tri_type);
    MPI_Type_commit(&upper_tri_type);

    if (my_rank == 0) {
        matrix = malloc(n * n * sizeof(int));
        printf("Digite os %d elementos (row-major):\n", n * n);
        for (int i = 0; i < n * n; i++) scanf("%d", &matrix[i]);
        MPI_Send(matrix, 1, upper_tri_type, 1, 0, MPI_COMM_WORLD);
        printf("Processo 0: triangular superior enviada.\n");
        free(matrix);
    } else if (my_rank == 1) {
        int *recv_buf = malloc(upper_count * sizeof(int));
        MPI_Recv(recv_buf, upper_count, MPI_INT, 0, 0,
                 MPI_COMM_WORLD, MPI_STATUS_IGNORE);
        printf("Processo 1 recebeu %d elementos:\n", upper_count);
        for (int i = 0; i < upper_count; i++) printf("%d ", recv_buf[i]);
        printf("\n");
        free(recv_buf);
    }

    MPI_Type_free(&upper_tri_type);
    free(blocklengths);
    free(displacements);
    MPI_Finalize();
    return 0;
}
